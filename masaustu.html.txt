    <script>
        var myMap, marker;
        let hidrantlar = JSON.parse(localStorage.getItem('itfaiye_db')) || [];
        ymaps.ready(() => {
            myMap = new ymaps.Map("map", { center: [37.4182, 31.8465], zoom: 14 });
            myMap.events.add('click', e => {
                let c = e.get('coords');
                document.getElementById('h-lat').value = c[0].toFixed(6);
                document.getElementById('h-lng').value = c[1].toFixed(6);
                if (marker) marker.geometry.setCoordinates(c);
                else { marker = new ymaps.Placemark(c); myMap.geoObjects.add(marker); }
            });
            listele();
        });
        function listele() {
            const tbody = document.getElementById('h-liste'); tbody.innerHTML = "";
            myMap.geoObjects.removeAll(); if(marker) myMap.geoObjects.add(marker);
            hidrantlar.forEach((h, i) => {
                tbody.innerHTML += `<tr><td>${h.ad}</td><td>${h.lat}, ${h.lng}</td>
                <td><span class="badge ${h.st=='aktif'?'bg-success':'bg-danger'}">${h.st.toUpperCase()}</span></td>
                <td><button onclick="sil(${i})" class="btn btn-sm btn-outline-danger">Sil</button></td></tr>`;
                myMap.geoObjects.add(new ymaps.Placemark([h.lat, h.lng], {balloonContent: h.ad}, {preset: h.st=='aktif'?'islands#blueWaterDropIcon':'islands#redWaterDropIcon'}));
            });
        }
        document.getElementById('h-form').onsubmit = e => {
            e.preventDefault();
            hidrantlar.push({ad: document.getElementById('h-adres').value, st: document.getElementById('h-durum').value, lat: document.getElementById('h-lat').value, lng: document.getElementById('h-lng').value});
            localStorage.setItem('itfaiye_db', JSON.stringify(hidrantlar));
            listele(); e.target.reset();
        };
        function sil(i) { if(confirm("Silinsin mi?")) { hidrantlar.splice(i,1); localStorage.setItem('itfaiye_db', JSON.stringify(hidrantlar)); listele(); } }
    </script>
